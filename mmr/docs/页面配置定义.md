## 页面定义
```json
{
  "id" : "<string>页面唯一标识",
  "version": "int 配置版本",
  "description": "页面说明",
  
  "init": {
    "@doc": "页面初始化行为",
    "actions": [
      {
        "name":"loadModel",
        "input": {
          "id":"student",
          "name": "student",
          "condition":"${student.no = @state.no}"
        }
      },
      {
          "name":"loadModels",
          "input": {
            "id":"classess",
            "name": "classes",
           "condition":"${classes.student = $student.id}"
          }
      }
    ]
  },
  "unload": {
    "@doc": "页面卸载行为"
  },

  "viewRoot": {
      "type": "layout", // 视图类型
      "@doc": "视图定义",
      "layout": {
          "@doc": "页面布局定义",
          "direction": "布局方向, 水平或者垂直",
          "itemSize": "子项大小, 宽或者高"
      },
      "items": [  // items 所有子项
        {
          "type": "input",
          "binding": "${student.no}",
          "attrs": {
            "label": "学号"
          }
        },
        {
          "type": "input",
          "binding": "${student.name}",
          "attrs": {
            "label": "姓名"
          }
        },
        {
           "type":"table",
           "attrs": {
              "binding": "${classes}",
              "columns": [
                {"header": "姓名", "type":"column", "binding": "${student.name}"},
                {"header": "成绩", "type":"column", "binding": "${classes.score}"},
                {"header": "操作", "type":"column", "binding": "${student.name}", "children":[
                        {"type":"button", "text":"提交", "action": {
                              "id":"submit-score",
                              "input": [
                                  "${student.id}",
                                  "${binding.item}"
                              ],
                              "output": [],
                              "post": {
                                 "refreshModel": {"id": "${classess}"}
                              }
                          }}
                ]}
              ]
           }        
        }
      ]
  }
}
```
